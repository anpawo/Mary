function print_c_or_d(int a, int b, char c, char d) -> int {
    if (a > b) {
        print(c);
    } else {
        print(d);
    }
}

operator ** precedence 8 (int n, int p) -> int {
    if (p == 0) {
        return 1
    } else {
        return n * n ** (p - 1)
    }
}

function helper(str message) -> void {
    print(message)
    exit(0)
}
