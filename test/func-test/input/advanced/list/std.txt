struct empty {}

struct list {
    any data;
    empty | list next;
}

/*

operator !! precedence 10 (list l, int index) -> null | any {
    if index < 0 then
    {
        return NULL;
    }
    else
    {
        if index == 0 then
        {
            return l;
        }
        else
        {
            return l . next !! (index - 1);
        }
    }
}
